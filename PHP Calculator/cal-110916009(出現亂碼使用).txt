            function displayNum($formula)
            {
               global $answer;
               global $negative;
               $x = (double)$formula;
               if($negative){
                  $x = (-1)*$x;
               }
               if((float)$answer == 0){
                  $answer = (string)$x;
               }
               else{
                  $answer = (string)((double)$answer*10 + $x);
               }
            }
            function displayFloat($formula)
            {
               global $negative;
               global $floatHow;
               global $answer;
               $y = (double)$formula;
               if ($negative) {
                    $y = (-1)*$y;
                }
               for ($i = 0; $i < $floatHow; $i++) {
                    $y = $y * 0.1;
               }
               $floatHow++;
               $answer = (string)round(((double)$answer + $y) ,8);
            }
